Récupérer les favoris de l'utilisateur

 CDVManager.sharedInstance().getFavoritesWithCompletion({ (error, favorites) in
                                if (error != nil) {
                                    NSLog("getFavoritesWithCompletion Error: \(error!.localizedDescription)")
                                } else {
                                    NSLog("\(favorites!.count) favorites for this user")
                                }
                            })

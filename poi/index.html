Créer un POI

Il est nécessaire de créer un objet de type CDVPOI puis de le sauver


 let newPOI = CDVPOI(dictionary: [
                    "name": "New POI",
                    "description": "Description of my POI  ",
                    "lat": 48.8834611,
                    "lng": 2.2652248,
                    "link": "bicycling",
                    "chrono_map": [
                        "time5",
                        "time10",
                        "time15"
                    ]
                ])

        CDVManager.sharedInstance().createPOI(newPOI) { (error, poiCreated) in
            if (error != nil) {
                NSLog("createPOI Error: \(error!.localizedDescription)")
            } else {
                NSLog("POI created")

            }}
